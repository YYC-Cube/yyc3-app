# 文件解析(同步)

> 创建文件解析任务，支持多种文件格式和解析工具。见 [文件解析服务](/cn/guide/tools/file–parser)

## OpenAPI

````yaml openapi/openapi.json post /paas/v4/files/parser/sync
paths:
  path: /paas/v4/files/parser/sync
  method: post
  servers:
    - url: https://open.bigmodel.cn/api/
      description: 开放平台服务
  request:
    security:
      - title: bearerAuth
        parameters:
          query: {}
          header:
            Authorization:
              type: http
              scheme: bearer
              description: >-
                使用以下格式进行身份验证：Bearer [<your api
                key>](https://bigmodel.cn/usercenter/proj-mgmt/apikeys)
          cookie: {}
    parameters:
      path: {}
      query: {}
      header: {}
      cookie: {}
    body:
      multipart/form-data:
        schemaArray:
          - type: object
            properties:
              file:
                allOf:
                  - type: string
                    format: binary
                    description: 待解析文件
              tool_type:
                allOf:
                  - type: string
                    enum:
                      - prime-sync
                    description: 使用的解析工具类型
              file_type:
                allOf:
                  - type: string
                    enum:
                      - WPS
                      - PDF
                      - DOCX
                      - DOC
                      - XLS
                      - XLSX
                      - PPT
                      - PPTX
                      - PNG
                      - JPG
                      - JPEG
                      - CSV
                      - TXT
                      - MD
                      - HTML
                      - BMP
                      - GIF
                      - WEBP
                      - HEIC
                      - EPS
                      - ICNS
                      - IM
                      - PCX
                      - PPM
                      - TIFF
                      - XBM
                      - HEIF
                      - JP2
                    description: >-
                      文件类型支持：pdf,docx,doc,xls,xlsx,ppt,pptx,png,jpg,jpeg,csv,txt,md,html,bmp,gif,webp,heic,eps,icns,im,pcx,ppm,tiff,xbm,heif,jp2
            required: true
            requiredProperties:
              - file
              - tool_type
        examples:
          example:
            value:
              tool_type: prime-sync
              file_type: WPS
  response:
    '200':
      application/json:
        schemaArray:
          - type: object
            properties:
              success:
                allOf:
                  - type: boolean
                    example: true
              message:
                allOf:
                  - type: string
                    example: 任务创建成功
              task_id:
                allOf:
                  - type: string
                    example: task_123456789
        examples:
          example:
            value:
              success: true
              message: 任务创建成功
              task_id: task_123456789
        description: 任务创建成功
    default:
      application/json:
        schemaArray:
          - type: object
            properties:
              error:
                allOf:
                  - required:
                      - code
                      - message
                    type: object
                    properties:
                      code:
                        type: string
                      message:
                        type: string
            refIdentifier: '#/components/schemas/Error'
        examples:
          example:
            value:
              error:
                code: <string>
                message: <string>
        description: 请求失败。
  deprecated: false
  type: path
components:
  schemas: {}

````
