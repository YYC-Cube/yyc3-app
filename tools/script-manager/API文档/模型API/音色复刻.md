# 音色复刻

> 使用音色复刻技术，基于示例音频生成指定音色、文本内容的语音合成。

## OpenAPI

````yaml openapi/openapi.json post /paas/v4/voice/clone
paths:
  path: /paas/v4/voice/clone
  method: post
  servers:
    - url: https://open.bigmodel.cn/api/
      description: 开放平台服务
  request:
    security:
      - title: bearerAuth
        parameters:
          query: {}
          header:
            Authorization:
              type: http
              scheme: bearer
              description: >-
                使用以下格式进行身份验证：Bearer [<your api
                key>](https://bigmodel.cn/usercenter/proj-mgmt/apikeys)
          cookie: {}
    parameters:
      path: {}
      query: {}
      header: {}
      cookie: {}
    body:
      application/json:
        schemaArray:
          - type: object
            properties:
              model:
                allOf:
                  - type: string
                    description: 模型
                    example: cogtts-clone
              voice_name:
                allOf:
                  - type: string
                    description: 指定唯一的音色名称
                    example: my_custom_voice_001
              text:
                allOf:
                  - type: string
                    description: 示例音频的文本内容，选填
                    example: 你好，这是一段示例音频的文本内容，用于音色复刻参考。
              input:
                allOf:
                  - type: string
                    description: 生成试听音频的目标文本内容
                    example: 欢迎使用我们的音色复刻服务，这将生成与示例音频相同音色的语音。
              file_id:
                allOf:
                  - type: string
                    description: 示例音频的file_id，通过文件接口上传获取
                    example: file_abc123def456ghi789
              request_id:
                allOf:
                  - type: string
                    description: 由用户端传递，需要唯一；用于区分每次请求的唯一标识符。如果用户端未提供，平台将默认生成。
                    example: voice_clone_req_001
            required: true
            refIdentifier: '#/components/schemas/VoiceCloneRequest'
            requiredProperties:
              - voice_name
              - input
              - file_id
              - model
        examples:
          example:
            value:
              model: cogtts-clone
              voice_name: my_custom_voice_001
              text: 你好，这是一段示例音频的文本内容，用于音色复刻参考。
              input: 欢迎使用我们的音色复刻服务，这将生成与示例音频相同音色的语音。
              file_id: file_abc123def456ghi789
              request_id: voice_clone_req_001
  response:
    '200':
      application/json:
        schemaArray:
          - type: object
            properties:
              voice:
                allOf:
                  - type: string
                    description: 音色
                    example: voice_clone_20240315_143052_001
              file_id:
                allOf:
                  - type: string
                    description: 音频试听文件ID
                    example: file_xyz789abc456def123
              file_purpose:
                allOf:
                  - type: string
                    description: 文件 purpose，固定为voice-clone-output
                    example: voice-clone-output
              request_id:
                allOf:
                  - type: string
                    description: 请求 ID
                    example: voice_clone_req_20240315_143052_001
            refIdentifier: '#/components/schemas/VoiceCloneResponse'
        examples:
          example:
            value:
              voice: voice_clone_20240315_143052_001
              file_id: file_xyz789abc456def123
              file_purpose: voice-clone-output
              request_id: voice_clone_req_20240315_143052_001
        description: 业务处理成功
    default:
      application/json:
        schemaArray:
          - type: object
            properties:
              error:
                allOf:
                  - required:
                      - code
                      - message
                    type: object
                    properties:
                      code:
                        type: string
                      message:
                        type: string
            refIdentifier: '#/components/schemas/Error'
        examples:
          example:
            value:
              error:
                code: <string>
                message: <string>
        description: 请求失败。
  deprecated: false
  type: path
components:
  schemas: {}
````
