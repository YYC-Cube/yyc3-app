# 解析结果

> 异步获取文件解析任务的结果，支持返回纯文本或下载链接格式。见 [文件解析服务](/cn/guide/tools/file–parser)

## OpenAPI

````yaml openapi/openapi.json get /paas/v4/files/parser/result/{taskId}/{format_type}
paths:
  path: /paas/v4/files/parser/result/{taskId}/{format_type}
  method: get
  servers:
    - url: https://open.bigmodel.cn/api/
      description: 开放平台服务
  request:
    security:
      - title: bearerAuth
        parameters:
          query: {}
          header:
            Authorization:
              type: http
              scheme: bearer
              description: >-
                使用以下格式进行身份验证：Bearer [<your api
                key>](https://bigmodel.cn/usercenter/proj-mgmt/apikeys)
          cookie: {}
    parameters:
      path:
        taskId:
          schema:
            - type: string
              required: true
              description: 文件解析任务ID
              example: task_123456789
        format_type:
          schema:
            - type: enum<string>
              enum:
                - text
                - download_link
              required: true
              description: 结果返回格式类型
              example: text
      query: {}
      header: {}
      cookie: {}
    body: {}
  response:
    '200':
      application/json:
        schemaArray:
          - type: object
            properties:
              status:
                allOf:
                  - type: string
                    enum:
                      - processing
                      - succeeded
                      - failed
                    description: 任务处理状态
                    example: succeeded
              message:
                allOf:
                  - type: string
                    description: 结果状态描述
                    example: 结果获取成功
              content:
                allOf:
                  - type: string
                    description: 当`format_type=text`时返回的解析文本内容
                    example: 这是解析后的文本内容…
                    nullable: true
              task_id:
                allOf:
                  - type: string
                    description: 文件解析任务`ID`
                    example: task_123456789
              parsing_result_url:
                allOf:
                  - type: string
                    description: 当`format_type=download_link`时返回的结果下载链接
                    example: https://example.com/download/result.zip
                    nullable: true
            refIdentifier: '#/components/schemas/FileParseResultResponse'
            requiredProperties:
              - status
              - message
              - task_id
        examples:
          example:
            value:
              status: succeeded
              message: 结果获取成功
              content: 这是解析后的文本内容…
              task_id: task_123456789
              parsing_result_url: https://example.com/download/result.zip
        description: 结果获取成功
    default:
      application/json:
        schemaArray:
          - type: object
            properties:
              error:
                allOf:
                  - required:
                      - code
                      - message
                    type: object
                    properties:
                      code:
                        type: string
                      message:
                        type: string
            refIdentifier: '#/components/schemas/Error'
        examples:
          example:
            value:
              error:
                code: <string>
                message: <string>
        description: 请求失败。
  deprecated: false
  type: path
components:
  schemas: {}

````
